rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    match /submissions/{weekId}/{fileName} {
      allow write: if request.resource.size < 6 * 1024 * 1024
                   && (request.resource.contentType.matches('application/pdf') ||
                       request.resource.contentType.matches('text/.*') ||
                       request.resource.contentType.matches('application/zip'));
      allow read: if true; // needed to generate URL after upload
    }

    match /registrants/{fileName} {
      allow write: if request.resource.size < 5 * 1024 * 1024;
      allow read: if true; // needed to access uploaded file URLs during registration
    }

    // Block everything else
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}
